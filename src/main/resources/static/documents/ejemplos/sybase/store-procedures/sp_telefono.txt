If @i_operacion = 'I'
begin
if @t_trn = 111
begin
  --ref20 Ini
  if @i_ente = 0
     begin
       /* 'No existe ente' */
       select @w_num = 101042
       goto mensaje_error
     end
  --ref20 Fin

  /* Verificacion de claves foraneas */
  if exists(select 1 from cobis..cl_producto where pd_producto = @i_producto)
     select @w_codigo = null
  else
     begin
       /* 'No existe producto' */
       select @w_num = 101032
       goto mensaje_error
     end

  if exists(select 1 from cobis..cl_direccion where di_ente = @i_ente and di_direccion = @i_direccion)
     select @w_codigo = null
  else
     begin
       /* 'No existe direccion' */
       select @w_num = 101001
       goto mensaje_error
     end

  exec @w_return = cobis..sp_catalogo
       @t_debug        = @t_debug,
       @t_file         = @t_file,
       @t_from         = @w_sp_name,
       @i_operacion    = 'E',
       @i_tabla        = 'cl_ttelefono',
       @i_codigo       = @i_tipo_telefono

  if @w_return != 0
     begin
       /* 'No existe tipo de telefono' */
       select @w_num = 101025
       goto mensaje_error
     end

  /* validar la region/prefijo del numero de telefono */
  select @w_region = substring(@i_valor, 1, 2)
  
  exec @w_return = cobis..sp_catalogo
    @t_debug      = @t_debug,
    @t_file       = @t_file,
    @t_from       = @w_sp_name,
    @i_operacion  = 'E',
    @i_tabla      = 'cl_region_telefono',
    @i_codigo     = @w_region

  if @w_return != 0
     begin
       /* 'NO EXISTE LA REGION DEL NUMERO DE TELEFONO' */
       select @w_num = 101329
       goto mensaje_error
     end

  /** obtener la longitud parametro de telefonos convencionales **/
  select @w_dig_cel = pa_tinyint
  from cobis..cl_parametro
  where pa_producto = 'MIS'
  and pa_nemonico = 'LONGT'		

  if datalength(@i_valor) != @w_dig_cel
     begin
       /* 'LONGITUD DEL NUMERO TELEFONICO INCORRECTA' */
       select @w_num = 101331
       goto mensaje_error
     end

  --ref20 Ini
  select @w_subtipo = en_subtipo,
         @w_fecha_crea_cli = convert(varchar(10), en_fecha_crea, 101),
         @w_idcliente  = isnull(en_ced_ruc,isnull(p_pasaporte,'')),
         @w_nombrecli  = isnull(en_nombre,''),
         @w_p_apellido = isnull(p_p_apellido,''),
         @w_s_apellido = isnull(p_s_apellido,'')
  from cobis..cl_ente
  where en_ente = @i_ente

     begin
       /* 'No existe ente' */
       select @w_num = 101042
       goto mensaje_error
     end
  --ref20 Fin

  if @w_subtipo = 'C' and @i_pertenece <> 'P'                
     begin
       /* 'para companias solo aplica propio en pertenece a'*/
       select @w_num = 101547
       goto mensaje_error
     end   

  --ref19 Valida si la direccion tipo domicilio tiene algun movil asociado
  if @w_subtipo = 'P' and @i_tipo_dir = 'D'
     begin
       if exists(select 1 from cl_direccion_email where de_ente = @i_ente and de_tipo = 'M' and de_dir_domici = @i_direccion)
          begin
            select @w_num = 101693
	    goto mensaje_error
          end      
     end

  if @i_no_cobis = 'S'

           @o_secuencial = @s_ssn out

          begin
            /* 'Error en creacion de transaccion de servicio'*/

            goto mensaje_error
          end

            if @i_producto = 2 and @i_canal <> 'VREFEREN'

    select @o_secuencial = isnull(max(te_secuencial),0) + 1
    from cobis..cl_telefono
    where te_ente = @i_ente
    and te_direccion = @i_direccion

         /* 'Error en creacion de telefono' */

         goto mensaje_error
       end

    /* Transaccion servicios - cl_telefono */

      secuencial, tipo_transaccion, alterno, clase, fecha,
      usuario, terminal, srv, lsrv, origen, rol, 
      ente, direccion, telefono, valor,
      tipo, extension, comentario, pertenece, canal)	--ref20
    values(@s_ssn, @t_trn, @p_alterno, 'N', @s_date,
      @s_user, @s_term, @s_srv, @s_lsrv, @i_producto, @s_rol, 
      @i_ente, @i_direccion, @o_secuencial, @i_valor,
      @i_tipo_telefono, @i_extension, @i_comentario, @i_pertenece, @w_canal)	--ref20

    if @@error != 0
       begin
         /* 'Error en creacion de transaccion de servicio' */
         select @w_num = 103005
         goto mensaje_error
       end

    --ref20 Ini
	if @i_envio_notifica = 'S' and @w_subtipo = 'P' 
       begin 
         if @w_fecha_reg > @w_fecha_crea_cli
            begin
              --Envia Notificacion al cliente por el cambio de telefono
              
              --Obtiene el email del cliente
     	      select top 1 @w_mail = de_descripcion
              from cobis..cl_direccion_email
              where de_ente = @i_ente
              and de_tipo = 'E'
              order by de_fecha_modificacion desc

              if @@rowcount = 0
                 select @w_mail = ''
			--ref21 Ini
			--Obtiene el celular del cliente
			select top 1 @w_celular = de_descripcion
				from cobis..cl_direccion_email
				where de_ente = @i_ente
				and de_tipo = 'M'
				order by de_fecha_modificacion desc

			if @@rowcount = 0
				select @w_celular = '' 
			--ref21 Fin
              select @w_servicio = 'Actualizacion de Telefono'
			  
      	      exec @w_return = cobis..sp_notif_cli_mod_dir
		 @i_producto 	   = @i_producto,
		 @i_ente     	   = @i_ente,
		 @i_canalmod       = @i_canal,
		 @i_oficina	   = @s_ofi,
		 @i_envio_notifica = @i_envio_notifica,	  --Cuando sean actualizaciones masivas se debe enviar el valor "N"
		 @i_idcliente      = @w_idcliente,
		 @i_today          = @w_today,
		 @i_nombrecli 	   = @w_nombrecli,
	 	 @i_p_apellido 	   = @w_p_apellido,
	 	 @i_s_apellido 	   = @w_s_apellido,
		 @i_servicio       = @w_servicio,
		 @i_mail           = @w_mail,
		 @i_telefono	   = '0'
			,@o_medio 			= @w_medio out,			--ref21
			@o_nombrecli 		= @w_nombrecliente out,	--ref21
			@o_fecha 			= @w_fecha out,			--ref21
			@o_hora 			= @w_hora out,			--ref21
			@o_idcli_enm 		= @w_idcli_enm out,		--ref21
			@o_nemonico 			= @w_nemonico out--ref
			
			  if @w_return != 0
			 begin
			   --'Error en NotificaciÃ³n al Cliente por ActualizaciÃ³n de Datos' 
			   select @w_num = @w_return
			   goto mensaje_error
			 end
				--ref21 Ini
				--Retorna campos para envÃ­o de notificaciÃ³n 
				select @w_medio, 
				@w_nombrecliente,
				@w_fecha,
				@w_hora,
				@w_idcli_enm,
				@w_servicio,
				@w_mail,
				@w_nemonico,
				@o_secuencial,
				@w_celular
				--ref21 Fin
            end
       end
    --ref20 Fin

  commit tran

  --select @o_secuencial

  return 0
end
else
begin
   /*  'No corresponde codigo de transaccion' */
   select @w_num = 151051
   goto mensaje_error
end
end
